# install dev environment
```bash
sudo apt update
# install compiler and tool, libs
sudo apt install build-essential

# install debugger
sudo apt install gdb
```


```bash
# check gcc version
gcc --version
# -> gcc (Ubuntu 9.4.0-1ubuntu1~20.04.2) 9.4.0

# check make version
make --version
# -> GNU Make 4.2.1

# check gdb verion
gdb --version
# -> GNU gdb (Ubuntu 9.2-0ubuntu1~20.04.1) 9.2
```
